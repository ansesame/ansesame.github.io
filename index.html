split_train_test <- function(auto, train_pct, SEED){
  set.seed(SEED)
  num_obs <- nrow(auto)
  train_idx <- sample(num_obs, num_obs*train_pct)
  test_idx <- seq(num_obs)[!seq(num_obs) %in% train_idx]
  
  train_data <- auto[train_idx,]
  test_data <- auto[test_idx,]  
  return(list('train_data' = train_data, 'test_data' = test_data))
}

Punto 2: knots en regresi√≥n spline.

A continuaci√≥n, mediante validaci√≥n cruzada en 10 folds, se determina el n√∫mero √≥ptimo de knots la regresi√≥n spline. Se prueban desde 1 hasta 10 knots, igualmente espaciados entre ellos.

## N√∫mero √≥ptimo de knots para el problema de regresi√≥n spline
optim_knots_spline_reg <- function(train_data, nfolds, max.knots, SEED){
  set.seed(SEED)
  rnd_sample = sample(1:nfolds,nrow(train_data),replace=TRUE) # Muestreo de folds
  table(rnd_sample)
  
  mse_knot <- NULL
  list_knot_val <- list()
  # Realizar validaci√≥n cruzada 
  for(nknots in 1:max.knots){
    # knots a testear
    seq.knots <- seq(min(train_data$horsepower), max(train_data$horsepower), 
                     length = nknots+2)[1:nknots+1]
    list_knot_val[[nknots]] <- seq.knots
    
    # Paso de validaci√≥n cruzada
    mse_sample <- NULL
    for(i in 1:nfolds){
      # i es la porcion de validaci√≥n
      mod_spl <- lm(mpg ~ bs(horsepower,knots = seq.knots, 
                             Boundary.knots = range(train_data$horsepower) + c(-5,5)), 
                    data=train_data[rnd_sample!=i,])
      pred_spl <- predict(mod_spl, train_data)[rnd_sample==i]
      mse_sample[i] <- ecm(train_data$mpg[rnd_sample==i], pred_spl)
    }
    mse_knot[nknots] <- mean(mse_sample)
    #print(paste('ECM promedio:', mean(mse_sample)))
    
  }
  best_nknots <- which.min(mse_knot)
  best_knots <- list_knot_val[[best_nknots]]
  return(list('nknots' = best_nknots, 'knots'= best_knots))
}
# Ejecuci√≥n

Punto 3: modelo basado en funciones

En este punto se compara el poder de predicci√≥n de tres modelos basados en funciones: polinomio grado 2 global, spline suavizado y del modelo de regresi√≥n spline √≥ptimo hallado en el punto anterior, mediante validaci√≥n cruzada.

best_model_functions_based <- function(train_data, nfolds, best_knots, SEED){
  set.seed(SEED)
  rnd_sample = sample(1:nfolds,nrow(train_data),replace=TRUE)
  table(rnd_sample)
  
  mse_poli_2 <- NULL
  mse_spli_suav <- NULL
  mse_spli_reg <- NULL
  for(i in 1:nfolds){
    # Polinomio grado 2 GLOBAL
    mod_poli_2 <- lm(mpg ~ poly(horsepower, 2), 
                     data=train_data[rnd_sample!=i,])
    pred_poli_2 <- predict(mod_poli_2, train_data)[rnd_sample==i]
    mse_poli_2[i] <- ecm(train_data$mpg[rnd_sample==i], pred_poli_2)
    
    # Spline suavizado
    mod_spli_suav <- smooth.spline(train_data[rnd_sample!=i,]$horsepower,
                                   train_data[rnd_sample!=i,]$mpg)
    pred_spli_suav <- predict(mod_spli_suav, train_data$horsepower)$y[rnd_sample==i]
    mse_spli_suav[i] <- ecm(train_data$mpg[rnd_sample==i], pred_spli_suav)
    
    # Regresi√≥n spline con knots √≥ptimos
    mod_spli_reg <- lm(mpg ~ bs(horsepower,knots = best_knots, # best knots 
                                Boundary.knots = range(train_data$horsepower) + c(-5,5)), 
                       data=train_data[rnd_sample!=i,])
    pred_spli_reg <- predict(mod_spli_reg, train_data)[rnd_sample==i]
    mse_spli_reg[i] <- ecm(train_data$mpg[rnd_sample==i], pred_spli_reg)
    
  }
  
  min_mse = min(c(mean(mse_poli_2), mean(mse_spli_suav), mean(mse_spli_reg)))
  if (mean(mse_poli_2) == min_mse){
    best_funct_model = 'poli_2'
  } else if (mean(mse_spli_suav) == min_mse){
    best_funct_model = 'spline_suav'
  } else {
    best_funct_model = 'spline_reg'
  }
  
  return(list('mse_poli_2' = mse_poli_2, 'mse_spli_suav' = mse_spli_suav,
              'mse_spli_reg' = mse_spli_reg, 'best_funct_model' = best_funct_model))
}

Punto 4: modelo basado en regresi√≥n local

best_model_local_reg <- function(train_data, nfolds, SEED){
  set.seed(SEED)
  rnd_sample = sample(1:nfolds,nrow(train_data),replace=TRUE)
  table(rnd_sample)
  
  mse_rloc_1 <- NULL
  mse_rloc_2 <- NULL
  for(i in 1:nfolds){
    # Regresi√≥n grado 1
    mod_rloc_1 <- loess(mpg ~ horsepower, data=train_data[rnd_sample!=i,],
                        family = "gaussian", degree = 1)
    pred_rloc_1 <- predict(mod_rloc_1, train_data)[rnd_sample==i]
    mse_rloc_1[i] <- ecm(train_data$mpg[rnd_sample==i], pred_rloc_1)
    
    # Regresi√≥n grado 2
    mod_rloc_2 <- loess(mpg ~ horsepower, data=train_data[rnd_sample!=i,],
                        family = "gaussian", degree = 2)
    pred_rloc_2 <- predict(mod_rloc_2, train_data)[rnd_sample==i]
    mse_rloc_2[i] <- ecm(train_data$mpg[rnd_sample==i], pred_rloc_2)
  }
  best_degree = if (mean(mse_rloc_1) < mean(mse_rloc_2)) 1 else 2
  return(list('mse_rloc_1' = mse_rloc_1, 'mse_rloc_2' = mse_rloc_2, 
              'best_degree' = best_degree))
}

best_mod_paradigm <- function(train_data, test_data, best_funct_model,  
                              best_gauss_degreee, best_knots=NULL, plot_figure=FALSE){
  # Train with all train data
  # Regresi√≥n polinomial local grado n
  mod_rloc <- loess(mpg ~ horsepower, data=train_data, family = "gaussian",
                      degree = best_gauss_degreee)
  pred_rloc <- predict(mod_rloc, test_data)
  mse_rloc <- ecm(test_data$mpg, pred_rloc)
  
  # Polinomio global grado 2
  mod_poli_2 <- lm(mpg ~ poly(horsepower, 2), data=train_data)
  pred_poli_2 <- predict(mod_poli_2, test_data)
  mse_poli_2 <- ecm(test_data$mpg, pred_poli_2)
  
  # Mejor modelo basado en base de funciones
  if (best_funct_model == 'spline_reg'){
    mod_func_based <- lm(mpg ~ bs(horsepower,knots = best_knots, 
                                  Boundary.knots = range(train_data$horsepower) + c(-5,5)), 
                         data=train_data)
    pred_func_based <- predict(mod_func_based, test_data)
    mse_func_based <- ecm(test_data$mpg, pred_func_based)
  } else if (best_funct_model == 'spline_suav'){
    mod_func_based <- smooth.spline(train_data$horsepower,
                                    train_data$mpg)
    pred_func_based <- predict(mod_func_based, train_data$horsepower)$y
    mse_func_based <- ecm(train_data$mpg, pred_func_based)
  } else {
    #Polinomio global grado 2 ya se calcula
    mse_func_based <- mse_poli_2
  }
  
  
  if (plot_figure){
    xvals = data.frame(horsepower = seq(min(train_data$horsepower),
                                        max(train_data$horsepower,by=1)))
    plot(train_data$horsepower,train_data$mpg,xlab="HP",ylab="mpg", pch=1)
    points(test_data$horsepower,test_data$mpg,xlab="HP",ylab="mpg", pch=4, col='blueviolet')
    pred <- predict(mod_func_based,xvals)
    lines(xvals$horsepower,pred,lwd=2,col="blue")
    pred2 <- predict(mod_rloc,xvals)
    lines(xvals$horsepower,pred2,lwd=2,col="red")
    pred3 <- predict(mod_poli_2,xvals)
    lines(xvals$horsepower,pred3,lwd=2,col="green")
    legend("topright", 
           c(paste('Spline knots:', round(mse_func_based, 3)),
             paste('Polinomial local grado 2:', round(mse_rloc, 3)),
             paste('Polinomial global grado 2:', round(mse_poli_2, 3)),
             paste('Datos de prueba')
           ),
           lty = c(1,1,1,0),
           pch = c('-','-','-','X'),
           col = c("blue", "red", "green", 'blueviolet'),
           title = 'ECM prueba modelo',
           cex = 0.7
    )
    title('Eligiendo entre paradigmas de modelamiento')
  }
  
  return(list('mse_func_based' = mse_func_based, 'mse_rloc' = mse_rloc,
              'mse_poli_2' = mse_poli_2))
  
}

Punto 1

En resumen, el estimador de Nadarya-Watson nos dice que \(x(t)\) es un promedio ponderado de las medidas de las \(n_i\) observaciones discretizadas que se tienen de la funci√≥n \(x_i\) en los puntos \(t_{i,1} ...t_{i,j}... t_{i,n_i}\). Esta ponderaci√≥n \(w(t, t_{i,j})\) que es una funci√≥n de kernel que dependede la distancia entre \(t\) y \(t_{t,i}\) y un par√°metro de suavizamiento. As√≠,

\[ x(t) = \frac{\sum_{j=1}^n_i w(t, t_{i,j}) x_{i,j}}{\sum_{j=1}^n_i w(t, t_{i,j})} \]
